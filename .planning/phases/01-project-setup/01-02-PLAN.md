```bash
npm install -D tailwindcss autoprefixer postcss
npx tailwindcss init -p
```

## Dark Mode Configuration (Pattern 4)
```javascript
// tailwind.config.js
export default {
  darkMode: 'class', // Manual control via .dark class
  content: [
    "./index.html",
    "./src/**/*.{vue,js,ts,jsx,tsx}",
  ],
  theme: {
    extend: {
      colors: {
        // Industrial dark mode palette
        gray: {
          850: '#1f2937',
          900: '#111827',
          950: '#030712',
        }
      }
    }
  }
}
```

## CSS Integration Pattern
```css
/* src/assets/main.css */
@tailwind base;
@tailwind components;
@tailwind utilities;

@layer base {
  :root {
    --color-bg-primary: #f8fafc;
    --color-bg-secondary: #ffffff;
  }

  .dark {
    --color-bg-primary: #0f172a;
    --color-bg-secondary: #1e293b;
  }
}
```

    # Step 3: Configure tailwind.config.js
    Replace the content of tailwind.config.js with:

    ```javascript
    /** @type {import('tailwindcss').Config} */
    export default {
      content: [
        "./index.html",
        "./src/**/*.{vue,js,ts,jsx,tsx}",
      ],
      darkMode: 'class', // Manual control via .dark class on HTML element
      theme: {
        extend: {
          colors: {
            // Industrial dark mode palette
            gray: {
              850: '#1f2937',
              900: '#111827',
              950: '#030712',
            },
            // Accent colors for industrial feel
            accent: {
              50: '#eff6ff',
              100: '#dbeafe',
              500: '#3b82f6',
              600: '#2563eb',
              700: '#1d4ed8',
            }
          }
        }
      },
      plugins: []
    }
    ```

    # Step 4: Verify postcss.config.js
    Ensure postcss.config.js contains:

    ```javascript
    export default {
      plugins: {
        tailwindcss: {},
        autoprefixer: {},
      },
    }
    ```

    # Step 5: Add Tailwind directives to main.css
    Replace the entire content of src/assets/main.css with:

    ```css
    @tailwind base;
    @tailwind components;
    @tailwind utilities;

    /* Custom dark mode styles */
    @layer base {
      :root {
        --color-bg-primary: #f8fafc;
        --color-bg-secondary: #ffffff;
        --color-text-primary: #0f172a;
      }

      .dark {
        --color-bg-primary: #0f172a;
        --color-bg-secondary: #1e293b;
        --color-text-primary: #f1f5f9;
      }
    }

    @layer components {
      /* Base button styles */
      .btn-primary {
        @apply px-4 py-2 bg-accent-600 text-white rounded-lg hover:bg-accent-700 transition-colors;
      }

      .dark .btn-primary {
        @apply bg-accent-500 hover:bg-accent-600;
      }
    }
    ```

    # Step 6: Update App.vue to test Tailwind
    Modify src/App.vue to include Tailwind test classes:

    ```vue

    <template>

          </h1>

          >
            Toggle Dark Mode

          <div class="mt-8 p-4 bg-white dark:bg-gray-800 rounded-lg shadow">
            <p class="text-sm text-gray-500 dark:text-gray-400">

            </p>
          </div>
        </div>
      </div>
    </template>
    ```
